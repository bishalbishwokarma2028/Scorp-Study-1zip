## AI PROVIDER FALLBACK SYSTEM — ai_provider.py

Create ai_provider.py with this exact logic. This file handles ALL AI calls. Never call any AI API directly anywhere else.

Fallback order:
Groq Key 1 → 2 → 3 → 4 → 5 → Gemini Key 1 → 2 → 3 → 4 → 5 → OpenAI → Hugging Face → Return error message

```python
import os
import requests

GROQ_KEYS = [os.environ.get(f"GROQ_API_KEY_{i}") for i in range(1,6) if os.environ.get(f"GROQ_API_KEY_{i}")]
GEMINI_KEYS = [os.environ.get(f"GEMINI_API_KEY_{i}") for i in range(1,6) if os.environ.get(f"GEMINI_API_KEY_{i}")]
OPENAI_KEY = os.environ.get("OPENAI_API_KEY")
HF_KEY = os.environ.get("HUGGINGFACE_API_KEY")

DEFAULT_SYSTEM = "You are Bishal's Assistant, an AI study helper built inside ScorpStudy by Bishal Bishwokarma. You are not ChatGPT, not Gemini, not an open-source model. You are ScorpStudy's assistant, created by Bishal Bishwokarma, a full-stack developer and IT student from Kathmandu, Nepal. Never claim to be made by OpenAI, Google, or Meta. Always say you were created by Bishal Bishwokarma."

def try_groq(prompt, system_prompt):
    for i, key in enumerate(GROQ_KEYS):
        try:
            res = requests.post(
                "https://api.groq.com/openai/v1/chat/completions",
                headers={"Authorization": f"Bearer {key}", "Content-Type": "application/json"},
                json={"model": "llama3-8b-8192", "messages": [{"role": "system", "content": system_prompt}, {"role": "user", "content": prompt}], "max_tokens": 2048},
                timeout=15
            )
            if res.status_code in [429, 401]:
                print(f"Groq key {i+1} failed ({res.status_code}), trying next...")
                continue
            data = res.json()
            text = data.get("choices", [{}])[0].get("message", {}).get("content")
            if text:
                print(f"Used: Groq Key {i+1}")
                return text
        except Exception as e:
            print(f"Groq key {i+1} error: {e}")
            continue
    return None

def try_gemini(prompt, system_prompt):
    for i, key in enumerate(GEMINI_KEYS):
        try:
            res = requests.post(
                f"https://generativelanguage.googleapis.com/v1beta/models/gemini-1.5-flash:generateContent?key={key}",
                headers={"Content-Type": "application/json"},
                json={"contents": [{"parts": [{"text": f"{system_prompt}\n\n{prompt}"}]}]},
                timeout=15
            )
            if res.status_code in [429, 401]:
                print(f"Gemini key {i+1} failed, trying next...")
                continue
            data = res.json()
            text = data.get("candidates", [{}])[0].get("content", {}).get("parts", [{}])[0].get("text")
            if text:
                print(f"Used: Gemini Key {i+1}")
                return text
        except Exception as e:
            print(f"Gemini key {i+1} error: {e}")
            continue
    return None

def try_openai(prompt, system_prompt):
    if not OPENAI_KEY:
        return None
    try:
        res = requests.post(
            "https://api.openai.com/v1/chat/completions",
            headers={"Authorization": f"Bearer {OPENAI_KEY}", "Content-Type": "application/json"},
            json={"model": "gpt-3.5-turbo", "messages": [{"role": "system", "content": system_prompt}, {"role": "user", "content": prompt}], "max_tokens": 2048},
            timeout=15
        )
        if not res.ok:
            return None
        data = res.json()
        text = data.get("choices", [{}])[0].get("message", {}).get("content")
        if text:
            print("Used: OpenAI GPT-3.5")
        return text
    except:
        return None

def try_huggingface(prompt):
    if not HF_KEY:
        return None
    try:
        res = requests.post(
            "https://api-inference.huggingface.co/models/mistralai/Mistral-7B-Instruct-v0.1",
            headers={"Authorization": f"Bearer {HF_KEY}", "Content-Type": "application/json"},
            json={"inputs": prompt, "parameters": {"max_new_tokens": 1024}},
            timeout=20
        )
        if not res.ok:
            return None
        data = res.json()
        text = data[0].get("generated_text") if isinstance(data, list) else None
        if text:
            print("Used: Hugging Face Mistral")
        return text
    except:
        return None

def ask_ai(prompt, system_prompt=None):
    if system_prompt is None:
        system_prompt = DEFAULT_SYSTEM

    result = try_groq(prompt, system_prompt)
    if result:
        return result

    result = try_gemini(prompt, system_prompt)
    if result:
        return result

    result = try_openai(prompt, system_prompt)
    if result:
        return result

    result = try_huggingface(f"{system_prompt}\n\n{prompt}")
    if result:
        return result

    return "AI is busy right now, please try again in a moment."
```

IMPORTANT: Every single AI call in the entire app must use ask_ai() from ai_provider.py. Never call Groq, Gemini, OpenAI or HuggingFace directly from any route.

```python
def normalize(text):
    return text.lower().strip().rstrip("?").rstrip(".")

def get_static_answer(question):
    q = normalize(question)
    if q in STATIC_CACHE:
        return STATIC_CACHE[q]
    if q in ALIASES:
        return STATIC_CACHE[ALIASES[q]]
    for alias, key in ALIASES.items():
        if alias in q:
            return STATIC_CACHE[key]
    return None

def make_cache_key(prompt):
    return hashlib.md5(normalize(prompt).encode()).hexdigest()

def load_cache():
    if os.path.exists(CACHE_FILE):
        with open(CACHE_FILE, "r") as f:
            return json.load(f)
    return {}

def save_cache(cache):
    with open(CACHE_FILE, "w") as f:
        json.dump(cache, f, indent=2)

def get_cached(prompt):
    static = get_static_answer(prompt)
    if static:
        return static
    cache = load_cache()
    key = make_cache_key(prompt)
    return cache.get(key)

def set_cache(prompt, answer):
    # Only cache if answer is substantial and not the error message
    if not answer or "AI is busy right now" in answer or len(answer) < 50:
        return
    cache = load_cache()
    key = make_cache_key(prompt)
    # Limit cache size to 500 entries
    if len(cache) > 500:
        oldest_keys = list(cache.keys())[:100]
        for k in oldest_keys:
            del cache[k]
    cache[key] = answer
    save_cache(cache)
```

In every Flask route that calls AI, check cache first:
```python
from cache_manager import get_cached, set_cache
from ai_provider import ask_ai

# Before calling AI:
cached = get_cached(user_prompt)
if cached:
    return jsonify({"response": cached, "from_cache": True})

# Call AI only if not cached:
response = ask_ai(user_prompt, system_prompt)
set_cache(user_prompt, response)
return jsonify({"response": response, "from_cache": False})
```

## DAILY USAGE LIMITS SYSTEM

Create a usage tracking system using Supabase. Every user has a daily limit of 30 AI queries per day. When the limit is reached, show: "Your today's free limit crossed. Try Again Tomorrow."

Image generation is separate: 3 images per day per user.

Add this SQL table to Supabase:
```sql
CREATE TABLE daily_usage (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid REFERENCES auth.users ON DELETE CASCADE,
  usage_date date DEFAULT CURRENT_DATE,
  ai_queries int DEFAULT 0,
  images_generated int DEFAULT 0,
  UNIQUE(user_id, usage_date)
);
ALTER TABLE daily_usage ENABLE ROW LEVEL SECURITY;
CREATE POLICY "own_usage" ON daily_usage FOR ALL USING (auth.uid() = user_id);
```

Create a Flask helper function check_and_increment_usage(user_id, usage_type):
- usage_type is either "ai_query" or "image"
- For ai_query: limit is 30 per day. Check daily_usage for today's date. If ai_queries >= 30, return False (blocked). Otherwise increment by 1 and return True.
- For image: limit is 3 per day. If images_generated >= 3, return False. Otherwise increment and return True.
- Use upsert to insert or update the row for today's date.
- Every AI call route must call this function before calling ask_ai().
- Every AI call that costs a query: Chat, Quiz Generator, Flashcard Maker, Summarizer, Notes AI Enhance, Notes AI Summarize, Translator, Mind Map, Formula Helper.
- When blocked, return JSON: {"error": "limit_reached", "message": "Your today's free limit crossed. Try Again Tomorrow."}
- On the frontend, when this error is received, show a styled modal or banner with this exact message: "Your today's free limit crossed. Try Again Tomorrow."
- Show remaining queries count on dashboard (e.g., "27 / 30 queries remaining today")
- Show remaining image count on image generator page (e.g., "2 / 3 images remaining today")

## SUPABASE DATABASE — Run this SQL in Supabase SQL Editor

```sql
CREATE TABLE notes (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid REFERENCES auth.users ON DELETE CASCADE,
  title text NOT NULL DEFAULT 'Untitled',
  content text DEFAULT '',
  tags text[] DEFAULT '{}',
  created_at timestamp DEFAULT now(),
  updated_at timestamp DEFAULT now()
);

CREATE TABLE chat_history (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid REFERENCES auth.users ON DELETE CASCADE,
  title text DEFAULT 'New Chat',
  subject text DEFAULT 'General',
  messages jsonb DEFAULT '[]',
  provider_used text,
  created_at timestamp DEFAULT now()
);

CREATE TABLE quiz_results (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid REFERENCES auth.users ON DELETE CASCADE,
  topic text,
  score int DEFAULT 0,
  total int DEFAULT 0,
  percentage float DEFAULT 0,
  difficulty text,
  question_type text,
  questions jsonb DEFAULT '[]',
  created_at timestamp DEFAULT now()
);

CREATE TABLE flashcards (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid REFERENCES auth.users ON DELETE CASCADE,
  topic text,
  cards jsonb DEFAULT '[]',
  created_at timestamp DEFAULT now()
);

CREATE TABLE summaries (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid REFERENCES auth.users ON DELETE CASCADE,
  source_type text DEFAULT 'text',
  original_text text,
  summary text,
  key_points jsonb DEFAULT '[]',
  exam_questions jsonb DEFAULT '[]',
  vocabulary jsonb DEFAULT '[]',
  created_at timestamp DEFAULT now()
);

CREATE TABLE generated_images (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid REFERENCES auth.users ON DELETE CASCADE,
  prompt text,
  enhanced_prompt text,
  image_url text,
  style text,
  created_at timestamp DEFAULT now()
);

CREATE TABLE mindmaps (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid REFERENCES auth.users ON DELETE CASCADE,
  topic text,
  map_data jsonb,
  created_at timestamp DEFAULT now()
);

CREATE TABLE translations (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid REFERENCES auth.users ON DELETE CASCADE,
  original_text text,
  translated_text text,
  source_language text,
  target_language text,
  created_at timestamp DEFAULT now()
);

CREATE TABLE daily_usage (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  user_id uuid REFERENCES auth.users ON DELETE CASCADE,
  usage_date date DEFAULT CURRENT_DATE,
  ai_queries int DEFAULT 0,
  images_generated int DEFAULT 0,
  UNIQUE(user_id, usage_date)
);

ALTER TABLE notes ENABLE ROW LEVEL SECURITY;
ALTER TABLE chat_history ENABLE ROW LEVEL SECURITY;
ALTER TABLE quiz_results ENABLE ROW LEVEL SECURITY;
ALTER TABLE flashcards ENABLE ROW LEVEL SECURITY;
ALTER TABLE summaries ENABLE ROW LEVEL SECURITY;
ALTER TABLE generated_images ENABLE ROW LEVEL SECURITY;
ALTER TABLE mindmaps ENABLE ROW LEVEL SECURITY;
ALTER TABLE translations ENABLE ROW LEVEL SECURITY;
ALTER TABLE daily_usage ENABLE ROW LEVEL SECURITY;

CREATE POLICY "own_notes" ON notes FOR ALL USING (auth.uid() = user_id);
CREATE POLICY "own_chats" ON chat_history FOR ALL USING (auth.uid() = user_id);
CREATE POLICY "own_quizzes" ON quiz_results FOR ALL USING (auth.uid() = user_id);
CREATE POLICY "own_flashcards" ON flashcards FOR ALL USING (auth.uid() = user_id);
CREATE POLICY "own_summaries" ON summaries FOR ALL USING (auth.uid() = user_id);
CREATE POLICY "own_images" ON generated_images FOR ALL USING (auth.uid() = user_id);
CREATE POLICY "own_mindmaps" ON mindmaps FOR ALL USING (auth.uid() = user_id);
CREATE POLICY "own_translations" ON translations FOR ALL USING (auth.uid() = user_id);
CREATE POLICY "own_usage" ON daily_usage FOR ALL USING (auth.uid() = user_id);
```
